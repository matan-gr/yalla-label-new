
# Stage 1: Build
FROM node:22-alpine AS build

# Capture API Key from Cloud Build arguments
ARG API_KEY
ENV VITE_API_KEY=$API_KEY

WORKDIR /app

# Install dependencies
COPY package.json package-lock.json* ./

# Update NPM to latest to ensure compatibility
RUN npm install -g npm@latest

# Use legacy-peer-deps to prevent potential conflicts with React 19 types/React 18 core
RUN npm install --legacy-peer-deps

# Copy source and build
COPY . .
RUN npm run build

# Stage 2: Serve
FROM nginx:alpine

# Clean default assets
RUN rm -rf /usr/share/nginx/html/*

# Copy artifacts
COPY --from=build /app/dist /usr/share/nginx/html
COPY nginx.txt /etc/nginx/conf.d/default.conf

# Permissions for non-root execution (Cloud Run requirement)
RUN touch /var/run/nginx.pid && \
    chown -R nginx:nginx /var/run/nginx.pid && \
    chown -R nginx:nginx /var/cache/nginx && \
    chown -R nginx:nginx /var/log/nginx && \
    chown -R nginx:nginx /etc/nginx/conf.d && \
    chown -R nginx:nginx /usr/share/nginx/html

USER nginx

EXPOSE 8080

HEALTHCHECK --interval=30s --timeout=3s \
  CMD curl -f http://localhost:8080/healthz || exit 1

CMD ["nginx", "-g", "daemon off;"]
